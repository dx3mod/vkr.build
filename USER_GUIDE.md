:root {
  /* Начала подсчёта страниц, включая заголовок. */

  --subheader-align: center;

  /* https://developer.mozilla.org/en-US/docs/Web/CSS/list-style-type */

  /* Шрифт кода. */

   /* Режим переноса фигур (рис, таблиц). По-умолчанию, если они не помещаются

}
